const SHEET_NAME = 'En Casa | Presencialidad'; function doOptions(e) { return ContentService.createTextOutput('').setMimeType(ContentService.MimeType.TEXT); } // Cambiá si tu hoja tiene otro nombre function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; const action = params.action; let result; if (action === 'guardar') { result = guardarRegistro(params); } else if (action === 'leer') { result = leerRegistros(params); } else if (action === 'leerColaborador') { result = leerColaborador(params); } else { result = { error: 'Acción no reconocida' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function guardarRegistro(p) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); // Verificar si ya existe un registro para este DNI y semana const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][1]) === String(p.dni) && data[i][3] === p.semana) { // Actualizar fila existente sheet.getRange(i + 1, 1, 1, 9).setValues([[ new Date().toISOString(), p.dni, p.nombre, p.semana, p.lun, p.mar, p.mie, p.jue, p.vie ]]); return { ok: true, accion: 'actualizado' }; } } // Agregar nueva fila sheet.appendRow([ new Date().toISOString(), p.dni, p.nombre, p.semana, p.lun, p.mar, p.mie, p.jue, p.vie ]); return { ok: true, accion: 'creado' }; } catch (err) { return { error: err.message }; } } function leerRegistros(p) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const semana = p.semana; const registros = []; for (let i = 1; i < data.length; i++) { if (!semana || data[i][3] === semana) { registros.push({ timestamp: data[i][0], dni: String(data[i][1]), nombre: data[i][2], semana: data[i][3], lun: data[i][4], mar: data[i][5], mie: data[i][6], jue: data[i][7], vie: data[i][8] }); } } return { ok: true, registros }; } catch (err) { return { error: err.message }; } } function leerColaborador(p) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const registros = []; for (let i = 1; i < data.length; i++) { if (String(data[i][1]) === String(p.dni)) { registros.push({ semana: data[i][3], lun: data[i][4], mar: data[i][5], mie: data[i][6], jue: data[i][7], vie: data[i][8] }); } } // Ordenar por semana descendente (más reciente primero) registros.sort((a, b) => b.semana.localeCompare(a.semana)); return { ok: true, registros }; } catch (err) { return { error: err.message }; } }
