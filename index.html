const SHEET_NAME = 'En Casa | Presencialidad'; function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return ss.getSheets()[0]; } return sheet; } function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; const action = params.action; let result; if (action === 'guardar') { result = guardarRegistro(params); } else if (action === 'leer') { result = leerRegistros(params); } else if (action === 'leerColaborador') { result = leerColaborador(params); } else { result = { error: 'Acción no reconocida' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function guardarRegistro(p) { try { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][1]).trim() === String(p.dni).trim() && data[i][3] === p.semana) { sheet.getRange(i + 1, 1, 1, 9).setValues([[ new Date().toISOString(), p.dni, p.nombre, p.semana, p.lun, p.mar, p.mie, p.jue, p.vie ]]); return { ok: true, accion: 'actualizado' }; } } sheet.appendRow([ new Date().toISOString(), p.dni, p.nombre, p.semana, p.lun, p.mar, p.mie, p.jue, p.vie ]); return { ok: true, accion: 'creado' }; } catch (err) { return { error: err.message }; } } function leerRegistros(p) { try { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); const semana = p.semana; const registros = []; for (let i = 1; i < data.length; i++) { if (!semana || data[i][3] === semana) { registros.push({ timestamp: data[i][0], dni: String(data[i][1]), nombre: data[i][2], semana: data[i][3], lun: data[i][4], mar: data[i][5], mie: data[i][6], jue: data[i][7], vie: data[i][8] }); } } return { ok: true, registros }; } catch (err) { return { error: err.message }; } } function leerColaborador(p) { try { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); let nombre = ''; const registros = []; for (let i = 1; i < data.length; i++) { if (String(data[i][1]).trim() === String(p.dni).trim()) { if (!nombre) nombre = data[i][2]; if (data[i][3]) { registros.push({ semana: data[i][3], lun: data[i][4], mar: data[i][5], mie: data[i][6], jue: data[i][7], vie: data[i][8] }); } } } if (!nombre) return { ok: false, error: 'DNI no encontrado' }; registros.sort((a, b) => b.semana.localeCompare(a.semana)); return { ok: true, nombre, registros }; } catch (err) { return { error: err.message }; } }
